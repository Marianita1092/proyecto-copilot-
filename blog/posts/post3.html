def bfs(grafo, inicio):
    visitados = set()
    cola = deque([inicio])
    resultado = []
    
    while cola:
        nodo = cola.popleft()
        
        if nodo in visitados:
            continue
        
        visitados.add(nodo)
        resultado.append(nodo)
        
        # Agregar vecinos a la cola

                    cola.append(vecino)
    
    return resultado

def dfs_iterativo(grafo, inicio):
    visitados = set()
    pila = [inicio]
    resultado = []
    
    while pila:
        nodo = pila.pop()
        
        if nodo in visitados:
            continue
        
        visitados.add(nodo)
        resultado.append(nodo)
        
        if nodo in grafo:
            for vecino in reversed(grafo[nodo]):
                if vecino not in visitados:
                    pila.append(vecino)
    
    return resultado

def existe_camino(grafo, inicio, destino):
    """Usa BFS para encontrar si existe un camino"""
    visitados = set()
    cola = deque([inicio])
    
    while cola:
        nodo = cola.popleft()
        if nodo == destino:
            return True
        if nodo in visitados:
            continue
        visitados.add(nodo)
        
        if nodo in grafo:
            cola.extend(grafo[nodo])
    
    return False

def contar_componentes(grafo):
    """Usa DFS para contar grupos de nodos conexos"""
    visitados = set()
    componentes = 0
    
    def dfs(nodo):
        visitados.add(nodo)

    return componentes
